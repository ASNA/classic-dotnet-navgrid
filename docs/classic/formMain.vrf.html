AVR Classic program to demonstrate OpnQryF v1.0

#

Using OpnQryF with AVR Classic requires the use of the QryFileName, QryKeyFlds, and QrySelect keywords. You must use ImpOpen(*No) with OpnQryF--open the file after assigning the QryKeyFlds and QrySelect values. Always use '*UNIQUE' for the QryFilename.

Event handlers

BegSr ioPositionTo KeyPress
#

Cap first key for position-to and call position-to on Enter key.

    DclSrParm KeyAscii TYPE(*INTEGER) LEN(2) BY(*REFERENCE)
#

Requires ioField named ioPositionTo.

 
    DclConst LOWER_A   Value(  97 )
    DclConst LOWER_Z   Value( 122 )
    DclConst CASE_DIFF Value(  32 )
    DclConst ENTER_KEY Value(  13 )
    DclConst EAT_KEY   Value(   0 )
#

Force first character only to upper case.

    If ( KeyAscii >= LOWER_A          ) AND  +
       ( KeyAscii <= LOWER_Z          ) AND  +
       ( ioPositionTo.SelStart = 0    )
        KeyAscii = KeyAscii - CASE_DIFF
    EndIf
 
    If ( KeyAscii = ENTER_KEY )
        KeyAscii = EAT_KEY
        sbfCust_PositionTo( ioPositionTo.Value )
        ioPositionTo.Value = *Blanks
    EndIf
EndSr

BegSr sbfCust_FillNext
#

Fill sbfCust with BufferSize records.

    DclFld BufferSize Type( *Integer ) Len( 4 )
    DclFld Idx        Type( *Integer ) Len( 4 )
    DclFld RowColor   Type( *Integer ) Len( 4 )
#

LeaveSr


    

  

  

  

    

      

        #
      

      
Number of records to put in subfile at one time.

    

    

      
    BufferSize = BUFFER
 
    Do FromVal(1) ToVal(BufferSize) Index(Idx)
        Read Cust
        If ( %EOF(Cust))
            LeaveSr
        EndIf
        sbfCust_RRN = sbfCust.RowCount + 1
        sbfCust_SetFields()
        Write sbfCust
 
        RowColor = SetRowColor(sbfCust_RRN - 1, RowColor)
    EndDo
EndSr
 
BegSr sbfCust_FillPrev

    

  

  

  

    

      

        #
      

      
Add previous x records to subfile.

    

    

      
    DclFld BegOfFile  Type( *Boolean )
    DclFld BufferSize Type( *Integer ) Len( 4 )
    DclFld NewRow     Type( *Integer ) Len( 4 )
    DclFld RecsRead   Type( *Integer ) Len( 4 )
    DclFld RowColor   Type( *Integer ) Len( 4 )
 

    LeaveSr
    Eval F2( RowColor = sbfCust.RecBackColor[ 0 ] ) Err( *Extended )
 
    BufferSize = BUFFER
    SetLL Cust Key( sbfCust_CMName )
    RecsRead = 0
    Do FromVal( 1 ) ToVal( BufferSize )
        ReadP Cust BOF( BegOfFile )
        If ( BegOfFile )
            Leave
        EndIf
 
        RecsRead    = RecsRead + 1
        NewRow      = sbfCust.AddItem( '', 0 )
        Chain sbfCust Key( NewRow + 1 )
        sbfCust_SetFields()
        Update sbfCust
 
        sbfCust_RRN = sbfCust.RowCount + 1
        RowColor    = SetRowColor( NewRow, RowColor )
    EndDo
 
    If ( RecsRead > 0 )
        sbfCust.Select( RecsRead, sbfCust.Col )
        sbfCust.FirstRow = RecsRead
    EndIf
EndSr
